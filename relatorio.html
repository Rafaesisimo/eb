Está abaixo o script de relatório

import time

print('Mantenha este programa aberto, assim você não precisará inserir todas as informações novamente. ')
print('==================================')

pres = ['**⩥ALISTAMENTO MILITAR⩤**\n============================================\n►**Instrutor(a):**','►**Auxiliar:** N/A','►**Patente:**','============================================►\n**Participantes:**\n ','►**Aprovados:** ','►**Observação:** N/A\n','============================================►\n**Data:** ','►**Horário:** ','===========================================\n===============**▼COMPROVAÇÃO▼**===============']

instrutor = input('Insira seu nome do Roblox: ')
patente = input('Insira sua patente, Ex: "[P-SGT]" ')
while True:
nomes = input('Insira o nome dos participantes, separando-os com vírgula: ')
print('==================================')
aprovados = input('Insira o nome dos aprovados, separando-os com vírgula: ')
print('==================================')
print('Carregando... ')
time.sleep(0.2)
listanomes = nomes.split(',')
listaaprovados = aprovados.split(',')

relatorio = ''
relatorio += pres[0]
relatorio += instrutor
relatorio += '\n'
relatorio += pres[1]
relatorio += '\n'
relatorio += pres[2]
relatorio += patente
relatorio += '\n'
relatorio += pres[3]
relatorio += '\n'
relatorio += str(listanomes).replace(',','\n').replace("'", "").replace('[','').replace(']','')
relatorio += '\n'
relatorio += pres[4]
relatorio += '\n'
relatorio += str(listaaprovados).replace(',','\n').replace("'", "").replace('[','').replace(']','')
relatorio += '\n'
relatorio += pres[5]
relatorio += '\n'
relatorio += pres[6]
relatorio += str(time.localtime().tm_mday) + '/' + str(time.localtime().tm_mon)+ '/' + str(time.localtime().tm_year)
relatorio += '\n'
relatorio += pres[7]
relatorio += str(time.localtime().tm_hour) + ':' + str(time.localtime().tm_min)
relatorio += '\n'
relatorio += pres[8]
relatorio += '\n'


print('==================================')
print()
print(relatorio)
print()
time.sleep(5)
input('Pressione enter para fazer outro relatório com as informações já salvas.' )
print()
print('==================================')
